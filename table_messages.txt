CREATE TABLE messages (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_email TEXT,
    source TEXT,
    room TEXT,
    task TEXT,
    requester TEXT,
    assignee TEXT,
    assigned_at DATETIME,
    link TEXT,
    source_ts TEXT,
    original_text TEXT,
    done BOOLEAN DEFAULT 0,
    is_deleted BOOLEAN DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    completed_at DATETIME,
    category TEXT DEFAULT 'todo',
    deadline TEXT,
    thread_id TEXT,
    assignee_reason TEXT,
    replied_to_id TEXT,
    is_context_query INTEGER DEFAULT 0,
    constraints TEXT DEFAULT '[]',
    metadata TEXT DEFAULT '{}',
    source_channels TEXT DEFAULT '[]'
);
CREATE INDEX idx_thread_id ON messages(thread_id);
CREATE INDEX idx_messages_dashboard_filter ON messages(user_email, is_deleted, done, category, assignee);
CREATE INDEX idx_messages_task ON messages (task);
CREATE INDEX idx_messages_room ON messages (room);
CREATE INDEX idx_messages_requester ON messages (requester);
CREATE INDEX idx_messages_assignee ON messages (assignee);
CREATE INDEX idx_messages_original_text ON messages (original_text);
CREATE INDEX idx_messages_source ON messages (source);
CREATE INDEX idx_messages_created_at_desc ON messages (created_at DESC);
CREATE INDEX idx_messages_user_email ON messages (user_email);
CREATE INDEX idx_messages_is_deleted ON messages (is_deleted);
CREATE INDEX idx_messages_completed_at ON messages (completed_at) WHERE done = 1;
CREATE UNIQUE INDEX idx_messages_user_source_ts ON messages(user_email, source_ts);
CREATE INDEX idx_messages_user_deleted_created ON messages (user_email, is_deleted, created_at DESC);
CREATE INDEX idx_messages_user_done_completed ON messages (user_email, done, completed_at DESC);
CREATE UNIQUE INDEX idx_user_ts ON messages(user_email, source_ts);
